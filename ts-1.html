Enums

                
enum UserRole {
    USER = 'user',
    MANAGER = 'manager',
    ADMIN = 'admin'
}

type UserData =  {
    name: string,
    surname: string,
    age: number,
    isAccepted: boolean,
    birthDate?: Date,
    roles: UserRole[]
}

const userData: UserData = {
    name: 'Some name',
    surname: 'Some surname',
    age: 21,
    isAccepted: true,
    birthDate: new Date(),
    roles: [UserRole.USER, UserRole.MANAGER, UserRole.ADMIN]
}

Generics

                
type UserData<T=number> =  {
    name: string,
    surname: string,
    age: T,
}

const firstUserData: UserData = {
    name: 'Some name',
    surname: 'Some surname',
    age: 21,
}

const secondUserData: UserData<string> = {
    name: 'Some name',
    surname: 'Some surname',
    age: '21 years',
}
